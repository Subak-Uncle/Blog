### List

```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8" />
    <title>List</title>
  </head>
  <body>
    <!-- 
	li : list
	ul : unordered(순서 없는) list
	ol : ordered(순서 있는) list 
	
	<parent - child>s
	ul - li
	ol - li 세트로 활용됨
-->
    <!-- 기본 형태 -->
    <ul>
      <li>coffee</li>
      <li>tea</li>
      <li>soda</li>
    </ul>

    <!-- 기본 형태랑 동일 -->
    <ul style="list-style-type: disc">
      <li>coffee</li>
      <li>tea</li>
      <li>soda</li>
    </ul>

    <ul style="list-style-type: square">
      <li>coffee</li>
      <li>tea</li>
      <li>soda</li>
    </ul>

    <ul style="list-style-type: circle">
      <li>coffee</li>
      <li>tea</li>
      <li>soda</li>
    </ul>

    <ul style="list-style-type: none">
      <li>coffee</li>
      <li>tea</li>
      <li>soda</li>
    </ul>

    <!--  ordered list -->

    <ol>
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
    </ol>

    <ol type="1">
      <!-- html4까지 type을 씀 -->
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
    </ol>

    <ol start="10">
      <!-- html4까지 type을 씀 -->
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
    </ol>

    <!-- 로마자 표기 -->
    <ol type="I">
      <!-- html4까지 type을 씀 -->
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
      <li>커피</li>
      <li>티</li>
      <li>음료수</li>
    </ol>

    <!-- ol, ul 혼합 -->
    <ul>
      <li>
        커피
        <ol type="1">
          <li>블랙</li>
          <li>밀크</li>
        </ol>
      </li>
      <li>티</li>
      <li>
        음료수
        <ol type="A">
          <li>콜라</li>
          <li>사이다</li>
        </ol>
      </li>
    </ul>
  </body>
</html>
```

### Table

```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8" />
    <title>table</title>
  </head>
  <body>
    <table>
      <!-- thead : 자동 가운데 정렬 -->
      <tr>
        <th>번호</th>
        <th>이름</th>
        <th>나이</th>
      </tr>
      <!-- </thead> -->

      <!-- <tbody> -->
      <tr>
        <td>1</td>
        <td>홍길동</td>
        <td>24</td>
      </tr>
      <!--  </tbody> -->
    </table>

    <!-- 원래 br 형식 -->
    <br /><br />

    <!-- 가운데 정렬 -->
    <div align="center">
      <!-- border : 테두리 굵기 -->
      <table border="1">
        <!-- column 크기 설정 -->
        <col width="50" />
        <col width="200" />
        <col width="100" />
        <tr>
          <th>번호</th>
          <th>이름</th>
          <th>나이</th>
          <!-- th는 자동 가운데 정렬  -->
        </tr>
        <tr>
          <td align="center">1</td>
          <td>홍길동</td>
          <td>24</td>
        </tr>
        <tr>
          <td align="center">2</td>
          <td>성춘향</td>
          <td>16</td>
        </tr>
      </table>
    </div>

    <br /><br />

    <!-- 테이블 제목 삽입 -->
    <table border="1">
      <!-- 테이블 안에 caption -> 자동 가운데 정렬 -->
      <caption>
        주소록
      </caption>
      <col width="50" />
      <col width="200" />
      <col width="100" />
      <!-- 열 병합 -->
      <tr>
        <th>번호</th>
        <th>이름</th>
        <th>나이</th>
        <th colspan="2">전화번호</th>
        <!-- <th>전화번호</th> -->
      </tr>
      <tr>
        <td align="center">1</td>
        <td>홍길동</td>
        <td>24</td>
        <td>123-4567</td>
        <td>010-1234-5678</td>
      </tr>
      <tr>
        <td align="center">2</td>
        <td>성춘향</td>
        <td>16</td>
        <td>234-5678</td>
        <td>010-2345-6789</td>
      </tr>
    </table>

    <br /><br />

    <!-- 행 병합 -->
    <table border="1">
      <tr>
        <th>번호</th>
        <td>1</td>
      </tr>
      <tr>
        <th>이름</th>
        <td>홍길동</td>
      </tr>
      <tr>
        <th rowspan="2">전화번호</th>
        <td>123-4567</td>
      </tr>
      <tr>
        <!-- <th>전화번호</th> -->
        <td>010-1234-5678</td>
      </tr>
    </table>
  </body>
</html>
```

### Form

```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8" />
    <title>sample3-index</title>
  </head>
  <body>
    <!--  link: 목적->이동 -->

    <a href="NewFile.html">NewFile.html로 이동</a>
    <br /><br />

    <a href="http://zum.com">Zum으로 이동</a>
    <br />

    <!-- 
	form : link의 목적
		   javascipt에서 접근하기 용이하다.
	
	값을 전달할 시에 사용되는 attribute
	id :	현재 페이지에서 하나만 인식. Javascript에서 접근할 목적
	class:	Javascript에서 접근할 목적. CSS에서 주로 사용한다.
	name:	(link 시) 값을 전달할 목적
-->

    <form action="NewFile.jsp">
      <!-- hidden : 값은 넘겨주지만 감추려면!! -->
      <input type="hidden" name="mynum" value="336" />

      아이디:<input type="text" name="id" placeholder="아이디입력" /><br />
      패스워드:<input type="text" name="pwd" /><br />

      <input type="submit" value="이동" />
      <input type="reset" value="초기화" />
    </form>
    <br /><br />

    <input type="text" placeholder="이름입력" />
    <br />

    <!-- button은 type 속성을 명시하지 않으면 submit과 같은 역할 -->
    <input type="button" value="버튼이름" />
    <br /><br />

    <!-- html5 -->
    <input type="date" />
    <br /><br />

    <input type="range" max="10" min="0" />
    <br /><br />

    <input type="search" placeholder="검색어" />
    <br /><br />

    <input type="color" value="#0000ff" />
  </body>
</html>
```

### CheckBox & Radio Button

```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8" />
    <title>CheckBox & Radio Button</title>
  </head>
  <body>
    CheckBox & Radio Button
    <br /><br />

    <!-- 
	ChexkBox : 독립적. on/off, 취미, 관심대상
	Radio Button : 세트로 구성.
 -->

    <form action="NewFile1.jsp">
      <!-- radio button : 주제로 안 묶으면 셋 다 선택이 됨 -->
      <input type="radio" value="benz" />벤츠<br />
      <input type="radio" value="bmw" />bmw<br />
      <input type="radio" value="saab" />샤브<br />
      <br />

      <!-- 하나만 선택 가능 ver / name만 가능. / class, id로는 안 묶임 -->
      <input type="radio" name="car" value="benz" checked="checked" />벤츠<br />
      <input type="radio" name="car" value="bmw" />bmw<br />
      <input type="radio" name="car" value="saab" />샤브<br />
      <br /><br />

      <input type="checkbox" name="hobby1" value="패션" />패션
      <!-- checked : 선택이 되어 있음 -->
      <input
        type="checkbox"
        name="hobby2"
        value="음악감상"
        checked="checked" />음악감상 
        <input type="checkbox" name="hobby3" value="게임" />게임
      <br /><br />

      <!-- select는 최소 하나 이상을 넘김 
 		multiple="multiple" 로 중복 선택 가능.	-->
      <select name="sCar" multiple="multiple">
        <option value="벤츠">benz</option>
        <option value="비엠더블유">bmw</option>
        <option value="사브">saab</option>
        <option value="도요타">toyota</option>
      </select>
      <br /><br />

      <input type="submit" value="전송" />
    </form>

    <br /><br />

    <!-- textarea는 외부에서 크기 조절도 가능 -->
    <textarea rows="10" cols="60" placeholder="자기소개 글 작성"></textarea>
    <br /><br />

    <video width="400" controls="controls">
      <source src="./mov_bbb.mp4" type="video/mp4" />
      <!-- .wav : 용량 문제로 사용x -->
    </video>
  </body>
</html>
```

```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
	pageEncoding="UTF-8"%>

<%
	String car = request.getParameter("car");

	System.out.println("자동차:" + car);

	String hobby[] = request.getParameterValues("hobby");

	if(hobby != null){
		for (String s : hobby) {
			System.out.println(s);
		}
	}

	String sCar = request.getParameter("sCar");

	System.out.println("선택 자동차:" + sCar);

	String carArr[] = request.getParameterValues("sCar");

	for(String s : carArr){
		System.out.println(s);
	}
%>
```
